// This file is the C startup stuff for Cuik
// it's been made as lightweight and self contained
// as possible.
// 

// kernel32:
char* GetCommandLineA();

void _start() {
	int argc = 0;
	char** argv = NULL;

	int code = main(argc, argv);

	ExitProcess(code);
}

// actual user entry point
int main(int argc, char** argv);
